[ 문제 1 ]

#include <stdio.h>

int sum(int n) {
  if (n == 1)
    return 1;
  return n + sum(n-1);
}

int main(void) {
  int N = 0;
  scanf("%d", &N);
  printf("%d", sum(N));
  return 0;
}


[ 문제 2 ]

#include <stdio.h>

void forward(int n) {
  if (n / 10 == 0)
    printf("%d\n", n);
  else {
    forward(n / 10);
    printf("%d\n", n % 10);
  }
}

int main(void) {
  int N = 0;
  scanf("%d", &N);
  forward(N);
  return 0;
}

#include <stdio.h>

void backward(int n) {
  if (n / 10 == 0)
    printf("%d", n);
  else {
    printf("%d\n", n % 10);
    backward(n / 10);
  }
}

int main(void) {
  int N = 0;
  scanf("%d", &N);
  backward(N);
  return 0;
}


[ 문제 4 ]

#include <stdio.h>

int maximum(int n, int max) {
  if (n <= 0)
    return max;
  int input;
  scanf("%d", &input);
  if (input > max)
    return maximum(n-1, input);
  else
    return maximum(n-1, max);
}

int main(void) {
  int N = 0;
  scanf("%d", &N);
  printf("%d", maximum(N, 0));
  return 0;
}


[ 문제 5 ]

#include <stdio.h>

void hanoi(char fr, char by, char to, int n) {
  if (n < 1)
    return;
  hanoi(fr, to, by, n-1);
  printf("%c %c\n", fr, to);
  hanoi(by, fr, to, n-1);
}

int main(void) {
  int N = 0;
  scanf("%d", &N);
  hanoi('A', 'B', 'C', N);
  return 0;
}


[ 문제 6 ]

#include <stdio.h>

int gcd(int a, int b) {
  if (b == 0)
    return a;
  return gcd(b, a % b);
}

int main(void) {
  int a = 0, b = 0;
  scanf("%d %d", &a, &b);
  printf("%d", gcd(a, b));
  return 0;
}


[ 문제 7 ]

#include <stdio.h>
#include <string.h>

int count(char * str, char ch, int i) {
  if (i == strlen(str))
    return 0;
  int num = 0;
  if (str[i] == ch)
    num = 1;
  return num += count(str, ch, i+1);
}

int main(void) {
  char str[100];
  char ch;
  scanf("%s %c", str, &ch);
  printf("%d", count(str, ch, 0));
  return 0;
}
